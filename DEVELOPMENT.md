## 🚀 Setup Development Environment

### 1. Prerequisites
- Node.js 18+ (check with `node --version`)
- Git
- VS Code (recommended) with extensions:
  - ESLint
  - Prettier (optional)
  - Tailwind CSS IntelliSense

### 2. Clone & Install

```bash
git clone https://github.com/pengadaan3210-prog/gedung-3210.git
cd gedung-3210
npm install
```

### 3. Environment Setup

```bash
cp .env.example .env
# Edit .env and add your Supabase credentials
```

### 4. Start Development Server

```bash
npm run dev
```

Visit `http://localhost:5173`

### 5. Optional: Debug in VS Code

Create `.vscode/launch.json`:

```json
{
  "version": "0.2.0",
  "configurations": [
    {
      "type": "chrome",
      "request": "attach",
      "name": "Attach to Chrome",
      "address": "localhost",
      "port": 9222,
      "pathMapping": {
        "/": "${workspaceRoot}/",
        "/src": "${workspaceRoot}/src"
      }
    }
  ]
}
```

## 🎨 Coding Standards

### TypeScript Rules

1. **Always type your props and return types**

```typescript
// ❌ Bad
const MyComponent = (props) => { ... }

// ✅ Good
interface MyComponentProps {
  title: string
  count?: number
  onClick?: (id: string) => void
}

const MyComponent = ({ title, count, onClick }: MyComponentProps) => { ... }
```

2. **Use proper type imports**

```typescript
// ❌ Bad
import { useState } from 'react'

// ✅ Good
import { type FC } from 'react'
import { useState } from 'react'
```

3. **Avoid `any` type**

```typescript
// ❌ Bad
const handleData = (data: any) => { ... }

// ✅ Good
const handleData = (data: SheetsData) => { ... }
```

### Naming Conventions

```typescript
// Components: PascalCase
const MyComponent = () => { ... }

// Functions/Hooks: camelCase
const fetchUserData = () => { ... }
const useUserData = () => { ... }

// Constants: UPPER_SNAKE_CASE
const MAX_RETRIES = 3
const API_BASE_URL = 'https://api.example.com'

// Boolean variables: prefix with is/has/can
const isLoading = false
const hasError = true
const canDelete = true
```

### Code Organization

```typescript
// 1. Imports (grouped and sorted)
import React from 'react'
import { useState } from 'react'
import { Card } from '@/components/ui/card'
import { useData } from '@/hooks'

// 2. Types
interface Props { ... }
type Status = 'idle' | 'loading' | 'error' | 'success'

// 3. Component
const MyComponent = ({ prop }: Props) => {
  // Hooks first
  const [state, setState] = useState<Status>('idle')
  const data = useData()
  
  // Computed values
  const filtered = data.filter(...)
  
  // Effects
  useEffect(() => { ... }, [])
  
  // Handlers
  const handleClick = () => { ... }
  
  // Render
  return (
    <div>...</div>
  )
}

// 4. Export
export default MyComponent
```

## 🧩 Component Guidelines

### Creating a New Component

1. **Create component file**

```bash
touch src/components/MyComponent.tsx
```

2. **Write component with types**

```typescript
import { type FC } from 'react'

interface MyComponentProps {
  title: string
  description?: string
}

const MyComponent: FC<MyComponentProps> = ({ title, description }) => {
  return (
    <div>
      <h1>{title}</h1>
      {description && <p>{description}</p>}
    </div>
  )
}

export default MyComponent
```

3. **Use Tailwind for styling**

```typescript
const MyComponent = () => (
  <div className="flex flex-col gap-4 p-6 bg-white rounded-lg shadow-md">
    <h1 className="text-xl font-bold">Title</h1>
  </div>
)
```

### Best Practices

- Keep components focused (single responsibility)
- Extract complex logic to custom hooks
- Use composition over inheritance
- Memoize expensive computations with `useMemo`
- Memoize callbacks with `useCallback`
- Use proper error boundaries for error handling

## 📊 Working with Data

### Using Custom Hooks

```typescript
import { useKegiatan } from '@/hooks/useSheetsData'

const MyPage = () => {
  const { data, isLoading, isError, refetch } = useKegiatan()
  
  if (isLoading) return <LoadingState />
  if (isError) return <ErrorState onRetry={refetch} />
  
  return (
    <div>
      {data.map(item => (
        <div key={item.id}>{item.uraianKegiatan}</div>
      ))}
    </div>
  )
}
```

### Caching & Refetching

```typescript
// Data is cached for 5 minutes by default
// Manual refetch
const { refetch } = useKegiatan()
await refetch()

// Force fresh data
await refetch({ force: true })
```

### Filtering & Sorting

Data filtering and sorting is handled in `DataTable.tsx`. Use it for consistent UI:

```typescript
<DataTable
  data={myData}
  onSelect={handleSelect}
/>
```

## ⚡ Performance Optimization

### 1. Code Splitting

Routes are automatically code-split by React Router.

### 2. Image Optimization

```typescript
// ❌ Bad - large images block rendering
<img src="/large-image.jpg" />

// ✅ Good - use next-gen formats or compress
<img src="/image.webp" alt="description" loading="lazy" />
```

### 3. Memoization

```typescript
import { useMemo, useCallback } from 'react'

const MyComponent = ({ items }: Props) => {
  // Memoize expensive computations
  const filtered = useMemo(
    () => items.filter(item => item.status === 'active'),
    [items]
  )
  
  // Memoize callbacks to prevent child re-renders
  const handleClick = useCallback((id: string) => {
    console.log('Clicked:', id)
  }, [])
  
  return <div>...</div>
}
```

### 4. React Query Best Practices

```typescript
// Don't refetch on window focus for less critical data
useQuery({
  queryKey: ['data'],
  queryFn: fetchData,
  refetchOnWindowFocus: false,
  staleTime: 5 * 60 * 1000, // 5 minutes
})

// Use select to extract only needed data
useQuery({
  queryKey: ['items'],
  queryFn: fetchItems,
  select: (data) => data.filter(item => item.active),
})
```

## 🔄 Common Patterns

### Loading States

```typescript
const { data, isLoading, isError, refetch } = useKegiatan()

if (isLoading) return <LoadingState />
if (isError) return <ErrorState onRetry={refetch} />

return <div>{/* render data */}</div>
```

### Error Handling

```typescript
try {
  const result = await fetchData()
  setState(result)
} catch (error) {
  if (error instanceof Error) {
    setError(error.message)
  } else {
    setError('An unexpected error occurred')
  }
}
```

### Modal/Dialog

```typescript
import { Dialog, DialogContent, DialogHeader } from '@/components/ui/dialog'

const [open, setOpen] = useState(false)

return (
  <>
    <Button onClick={() => setOpen(true)}>Open</Button>
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogContent>
        <DialogHeader>Title</DialogHeader>
        <p>Content</p>
      </DialogContent>
    </Dialog>
  </>
)
```

### Form Handling

```typescript
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'

const schema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
})

const { register, handleSubmit, formState: { errors } } = useForm({
  resolver: zodResolver(schema),
})

const onSubmit = (data: z.infer<typeof schema>) => {
  console.log(data)
}

return (
  <form onSubmit={handleSubmit(onSubmit)}>
    <input {...register('email')} />
    {errors.email && <span>{errors.email.message}</span>}
  </form>
)
```
